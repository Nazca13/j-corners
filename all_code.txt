===== ./src/app/admin/page.js =====
'use client'
import { useEffect, useState, useRef } from 'react'
import { useRouter } from 'next/navigation'
import { supabase } from '@/lib/supabase'

// ─── CONSTANTS ───────────────────────────────────────────────────────────────
const M = '#8B1A1A'
const M2 = '#6B1313'
const BG = '#F5F0ED'
const SIDEBAR_W = 240

const STATUS = {
  Menunggu: { bg: '#FFF8E1', text: '#E65100', dot: '#FFB300', icon: '⏳' },
  Diproses: { bg: '#E3F2FD', text: '#1565C0', dot: '#2196F3', icon: '🔥' },
  Diantar:  { bg: '#EDE7F6', text: '#4527A0', dot: '#7C4DFF', icon: '🛵' },
  Selesai:  { bg: '#E8F5E9', text: '#2E7D32', dot: '#43A047', icon: '✅' },
}
const FLOW = ['Menunggu', 'Diproses', 'Diantar', 'Selesai']
const CATS = ['Food', 'Bakery', 'Coffee', 'Non Coffee']

const NAV = [
  { id: 'dashboard', label: 'Dashboard',     icon: '◉' },
  { id: 'orders',    label: 'Pesanan',        icon: '📋' },
  { id: 'menu',      label: 'Kelola Menu',    icon: '🍽️' },
  { id: 'settings',  label: 'Pengaturan',     icon: '⚙️' },
]

// ─── HELPERS ─────────────────────────────────────────────────────────────────
const fmtRp  = n  => `Rp ${(n || 0).toLocaleString('id-ID')}`
const fmtDate = d => new Date(d).toLocaleDateString('id-ID', { day:'numeric', month:'short', hour:'2-digit', minute:'2-digit' })

// ─── MAIN COMPONENT ──────────────────────────────────────────────────────────
export default function AdminPage() {
  const router  = useRouter()
  const [page, setPage]     = useState('dashboard')
  const [loading, setLoading] = useState(true)
  const [sideOpen, setSideOpen] = useState(false)   // mobile sidebar

  // data
  const [orders,   setOrders]   = useState([])
  const [products, setProducts] = useState([])
  const [user,     setUser]     = useState(null)

  // pesanan
  const [updatingId, setUpdatingId] = useState(null)
  const [filterStatus, setFilterStatus] = useState('all')
  const [searchOrder,  setSearchOrder]  = useState('')

  // modal menu
  const [modal,  setModal]  = useState(false)
  const [saving, setSaving] = useState(false)
  const [form,   setForm]   = useState({ id: null, name: '', category: 'Food', price: '', description: '', image: null, existing_image: '', available: true })
  const [menuFilter, setMenuFilter] = useState('all')

  // ─── FETCH ────────────────────────────────────────────────────────────────
  const fetchAll = async () => {
    setLoading(true)
    const { data: { session } } = await supabase.auth.getSession()
    if (!session) { router.replace('/login'); return }
    setUser(session.user)

    const [{ data: od }, { data: pd }] = await Promise.all([
      supabase.from('orders').select('*').order('created_at', { ascending: false }),
      supabase.from('products').select('*').order('created_at', { ascending: false }),
    ])
    setOrders(od   || [])
    setProducts(pd || [])
    setLoading(false)
  }

  useEffect(() => { fetchAll() }, [])

  // ─── ORDERS ───────────────────────────────────────────────────────────────
  const updateStatus = async (id, newStatus) => {
    setUpdatingId(id)
    const { error } = await supabase.from('orders').update({ status: newStatus }).eq('id', id)
    if (!error) setOrders(prev => prev.map(o => o.id === id ? { ...o, status: newStatus } : o))
    setUpdatingId(null)
  }

  const filteredOrders = orders.filter(o => {
    const matchStatus = filterStatus === 'all' || o.status === filterStatus
    const matchSearch = !searchOrder || o.nama_pembeli?.toLowerCase().includes(searchOrder.toLowerCase())
    return matchStatus && matchSearch
  })

  // ─── MENU CRUD ────────────────────────────────────────────────────────────
  const openModal = (prod = null) => {
    setForm(prod
      ? { id: prod.id, name: prod.name, category: prod.category, price: prod.price, description: prod.description || '', image: null, existing_image: prod.image_url || '', available: prod.available !== false }
      : { id: null, name: '', category: 'Food', price: '', description: '', image: null, existing_image: '', available: true }
    )
    setModal(true)
  }

  const handleDelete = async (id) => {
    if (!confirm('Yakin hapus menu ini?')) return
    await supabase.from('products').delete().eq('id', id)
    setProducts(prev => prev.filter(p => p.id !== id))
  }

  const handleSave = async (e) => {
    e.preventDefault()
    setSaving(true)
    let imageUrl = form.existing_image

    if (form.image) {
      const ext   = form.image.name.split('.').pop()
      const fname = `menu_${Date.now()}.${ext}`
      const { error } = await supabase.storage.from('product-images').upload(fname, form.image)
      if (!error) imageUrl = supabase.storage.from('product-images').getPublicUrl(fname).data.publicUrl
      else alert('Gagal upload gambar!')
    }

    const payload = { name: form.name, category: form.category, price: parseInt(form.price), description: form.description, image_url: imageUrl, available: form.available }

    if (form.id) {
      const { data, error } = await supabase.from('products').update(payload).eq('id', form.id).select()
      if (!error) setProducts(prev => prev.map(p => p.id === form.id ? data[0] : p))
    } else {
      const { data, error } = await supabase.from('products').insert([payload]).select()
      if (!error) setProducts(prev => [data[0], ...prev])
    }

    setSaving(false)
    setModal(false)
  }

  const filteredProducts = menuFilter === 'all' ? products : products.filter(p => p.category === menuFilter)

  // ─── STATS ────────────────────────────────────────────────────────────────
  const totalRevenue = orders.filter(o => o.status === 'Selesai').reduce((s, o) => s + (o.total_harga || 0), 0)
  const todayOrders  = orders.filter(o => new Date(o.created_at).toDateString() === new Date().toDateString()).length

  const logout = async () => {
    await supabase.auth.signOut()
    router.replace('/login')
  }

  // ─── STYLES ───────────────────────────────────────────────────────────────
  const card = { background: '#fff', borderRadius: 20, boxShadow: '0 1px 3px rgba(0,0,0,0.05),0 8px 24px rgba(0,0,0,0.06)', overflow: 'hidden' }
  const inp  = { width: '100%', padding: '11px 14px', border: '1.5px solid #EEE4E1', borderRadius: 12, fontFamily: 'Poppins,sans-serif', fontSize: 13, outline: 'none', background: '#FDFAF9', marginBottom: 12, color: '#1A0A0A' }

  // ─── LOADING ──────────────────────────────────────────────────────────────
  if (loading) return (
    <div style={{ minHeight: '100vh', background: BG, display: 'flex', flexDirection: 'column', alignItems: 'center', justifyContent: 'center', fontFamily: 'Poppins,sans-serif', gap: 16 }}>
      <div style={{ width: 48, height: 48, border: `4px solid #F0E6E6`, borderTopColor: M, borderRadius: '50%', animation: 'spin 0.8s linear infinite' }} />
      <div style={{ fontSize: 14, color: '#888', fontWeight: 600 }}>Memuat dapur J-Corners...</div>
      <style>{`@keyframes spin{to{transform:rotate(360deg)}}`}</style>
    </div>
  )

  // ─── SIDEBAR ──────────────────────────────────────────────────────────────
  const Sidebar = ({ mobile = false }) => (
    <aside style={{
      width: mobile ? '100%' : SIDEBAR_W,
      background: `linear-gradient(175deg, #1A0A0A 0%, ${M2} 60%, #8B1A1A 100%)`,
      minHeight: mobile ? 'auto' : '100vh',
      display: 'flex', flexDirection: 'column',
      padding: mobile ? '20px 16px 24px' : '28px 0 24px',
      position: mobile ? 'relative' : 'fixed',
      top: 0, left: 0, zIndex: 200,
      boxShadow: mobile ? 'none' : '4px 0 24px rgba(0,0,0,0.12)',
    }}>
      {/* Logo */}
      <div style={{ padding: mobile ? 0 : '0 24px', marginBottom: 32, display: 'flex', alignItems: 'center', gap: 10 }}>
        <div style={{ width: 36, height: 36, background: 'rgba(255,255,255,0.15)', borderRadius: 10, display: 'flex', alignItems: 'center', justifyContent: 'center', backdropFilter: 'blur(8px)' }}>
          <img src="/icons/logo-web.svg" alt="J" style={{ height: 20, filter: 'brightness(0) invert(1)' }} />
        </div>
        <div>
          <div style={{ fontSize: 15, fontWeight: 800, color: '#fff', letterSpacing: '-0.3px', lineHeight: 1 }}>J-Corners</div>
          <div style={{ fontSize: 10, color: 'rgba(255,255,255,0.5)', marginTop: 2 }}>Owner Panel</div>
        </div>
      </div>

      {/* Nav */}
      <nav style={{ flex: 1 }}>
        {NAV.map(n => {
          const active = page === n.id
          return (
            <button key={n.id} onClick={() => { setPage(n.id); setSideOpen(false) }} style={{
              display: 'flex', alignItems: 'center', gap: 12,
              width: '100%', padding: mobile ? '12px 16px' : '12px 24px',
              background: active ? 'rgba(255,255,255,0.15)' : 'transparent',
              border: 'none', borderRadius: active ? (mobile ? 12 : 0) : 0,
              borderLeft: active && !mobile ? '3px solid #fff' : mobile ? 'none' : '3px solid transparent',
              color: active ? '#fff' : 'rgba(255,255,255,0.55)',
              fontSize: 13, fontWeight: active ? 700 : 500,
              fontFamily: 'Poppins,sans-serif',
              cursor: 'pointer', transition: 'all 0.2s', textAlign: 'left',
              marginBottom: 2,
            }}>
              <span style={{ fontSize: 16 }}>{n.icon}</span>
              {n.label}
              {n.id === 'orders' && orders.filter(o => o.status === 'Menunggu').length > 0 && (
                <span style={{ marginLeft: 'auto', background: '#FFB300', color: '#fff', fontSize: 10, fontWeight: 800, padding: '2px 7px', borderRadius: 20, minWidth: 20, textAlign: 'center' }}>
                  {orders.filter(o => o.status === 'Menunggu').length}
                </span>
              )}
            </button>
          )
        })}
      </nav>

      {/* User + Logout */}
      <div style={{ padding: mobile ? '16px 0 0' : '16px 20px 0', borderTop: '1px solid rgba(255,255,255,0.1)', marginTop: 16 }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 12 }}>
          <div style={{ width: 34, height: 34, borderRadius: '50%', background: 'rgba(255,255,255,0.2)', display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: 14 }}>👤</div>
          <div>
            <div style={{ fontSize: 12, fontWeight: 700, color: '#fff' }}>Owner</div>
            <div style={{ fontSize: 10, color: 'rgba(255,255,255,0.5)', overflow: 'hidden', textOverflow: 'ellipsis', whiteSpace: 'nowrap', maxWidth: 120 }}>{user?.email}</div>
          </div>
        </div>
        <button onClick={logout} style={{ width: '100%', padding: '9px 0', background: 'rgba(255,255,255,0.1)', border: '1px solid rgba(255,255,255,0.15)', borderRadius: 10, color: 'rgba(255,255,255,0.7)', fontSize: 12, fontWeight: 700, fontFamily: 'Poppins,sans-serif', cursor: 'pointer', transition: 'background 0.2s' }}>
          🚪 Keluar
        </button>
      </div>
    </aside>
  )

  // ─── VIEWS ────────────────────────────────────────────────────────────────

  // DASHBOARD VIEW
  const DashboardView = () => (
    <div>
      <SectionHeader title="Dashboard" subtitle={`Selamat datang kembali! Berikut ringkasan hari ini.`} action={<RefreshBtn />} />

      {/* Stats Grid */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(180px,1fr))', gap: 14, marginBottom: 24 }}>
        {[
          { label: 'Total Pendapatan', value: fmtRp(totalRevenue), icon: '💰', color: '#2E7D32', bg: '#E8F5E9', sub: 'Dari pesanan selesai' },
          { label: 'Pesanan Hari Ini', value: todayOrders,         icon: '📦', color: '#1565C0', bg: '#E3F2FD', sub: 'Masuk hari ini' },
          { label: 'Total Menu',       value: products.length,     icon: '🍽️', color: M,        bg: '#FFF4F4', sub: 'Item di katalog' },
          { label: 'Menunggu Konfirm', value: orders.filter(o=>o.status==='Menunggu').length, icon: '⏳', color: '#E65100', bg: '#FFF8E1', sub: 'Perlu tindakan' },
        ].map((s, i) => (
          <div key={i} style={{ ...card, padding: '20px 20px 18px' }}>
            <div style={{ width: 42, height: 42, background: s.bg, borderRadius: 12, display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: 20, marginBottom: 14 }}>{s.icon}</div>
            <div style={{ fontSize: 24, fontWeight: 800, color: '#1A0A0A', letterSpacing: '-0.5px' }}>{s.value}</div>
            <div style={{ fontSize: 12, fontWeight: 700, color: '#888', marginTop: 4 }}>{s.label}</div>
            <div style={{ fontSize: 11, color: '#bbb', marginTop: 2 }}>{s.sub}</div>
          </div>
        ))}
      </div>

      {/* Status Pipeline */}
      <div style={{ ...card, padding: '20px 22px', marginBottom: 24 }}>
        <div style={{ fontSize: 14, fontWeight: 700, color: '#1A0A0A', marginBottom: 16 }}>Pipeline Pesanan</div>
        <div style={{ display: 'flex', gap: 12 }}>
          {FLOW.map((s, i) => {
            const cnt = orders.filter(o => o.status === s).length
            const sc  = STATUS[s]
            return (
              <div key={s} style={{ flex: 1, background: sc.bg, borderRadius: 14, padding: '16px 14px', textAlign: 'center', position: 'relative', cursor: 'pointer', transition: '0.2s' }}
                onClick={() => { setPage('orders'); setFilterStatus(s) }}>
                <div style={{ fontSize: 24 }}>{sc.icon}</div>
                <div style={{ fontSize: 28, fontWeight: 900, color: sc.text, margin: '6px 0 2px', letterSpacing: '-1px' }}>{cnt}</div>
                <div style={{ fontSize: 10, fontWeight: 700, color: sc.text, opacity: 0.8 }}>{s}</div>
                {i < FLOW.length - 1 && <span style={{ position: 'absolute', right: -8, top: '50%', transform: 'translateY(-50%)', fontSize: 16, color: '#ccc' }}>›</span>}
              </div>
            )
          })}
        </div>
      </div>

      {/* Pesanan Terbaru */}
      <div style={{ ...card }}>
        <div style={{ padding: '18px 22px 14px', display: 'flex', alignItems: 'center', justifyContent: 'space-between', borderBottom: '1px solid #F5EEEE' }}>
          <div style={{ fontSize: 14, fontWeight: 700, color: '#1A0A0A' }}>Pesanan Terbaru</div>
          <button onClick={() => setPage('orders')} style={{ fontSize: 12, color: M, fontWeight: 700, background: 'none', border: 'none', cursor: 'pointer' }}>Lihat Semua →</button>
        </div>
        {orders.slice(0, 5).map((o, i) => {
          const sc = STATUS[o.status] || STATUS.Menunggu
          return (
            <div key={o.id} style={{ padding: '14px 22px', display: 'flex', alignItems: 'center', gap: 14, borderBottom: i < 4 ? '1px solid #F9F4F3' : 'none' }}>
              <div style={{ width: 36, height: 36, background: sc.bg, borderRadius: 10, display: 'flex', alignItems: 'center', justifyContent: 'center', fontSize: 16, flexShrink: 0 }}>{sc.icon}</div>
              <div style={{ flex: 1, minWidth: 0 }}>
                <div style={{ fontSize: 13, fontWeight: 700, color: '#1A0A0A' }}>{o.nama_pembeli || '—'}</div>
                <div style={{ fontSize: 11, color: '#aaa', marginTop: 1 }}>#{o.id?.slice(0,8).toUpperCase()} · {fmtDate(o.created_at)}</div>
              </div>
              <div style={{ textAlign: 'right', flexShrink: 0 }}>
                <div style={{ fontSize: 14, fontWeight: 800, color: M }}>{fmtRp(o.total_harga)}</div>
                <span style={{ fontSize: 10, fontWeight: 700, background: sc.bg, color: sc.text, padding: '2px 8px', borderRadius: 20 }}>{o.status}</span>
              </div>
            </div>
          )
        })}
        {orders.length === 0 && <div style={{ padding: '40px 0', textAlign: 'center', color: '#bbb', fontSize: 13 }}>Belum ada pesanan</div>}
      </div>
    </div>
  )

  // ORDERS VIEW
  const OrdersView = () => (
    <div>
      <SectionHeader title="Pesanan" subtitle={`${orders.length} total pesanan masuk`} action={<RefreshBtn />} />

      {/* Filter Bar */}
      <div style={{ display: 'flex', gap: 8, marginBottom: 16, flexWrap: 'wrap' }}>
        {['all', ...FLOW].map(s => (
          <button key={s} onClick={() => setFilterStatus(s)} style={{
            padding: '7px 16px', borderRadius: 20, fontSize: 12, fontWeight: 700,
            fontFamily: 'Poppins,sans-serif', border: 'none', cursor: 'pointer', transition: '0.2s',
            background: filterStatus === s ? M : '#fff',
            color: filterStatus === s ? '#fff' : '#888',
            boxShadow: filterStatus === s ? `0 4px 12px rgba(139,26,26,0.25)` : '0 1px 4px rgba(0,0,0,0.06)',
          }}>
            {s === 'all' ? '📋 Semua' : `${STATUS[s].icon} ${s}`}
            {s !== 'all' && <span style={{ marginLeft: 6, background: filterStatus === s ? 'rgba(255,255,255,0.25)' : '#f0e6e6', padding: '1px 6px', borderRadius: 10 }}>{orders.filter(o=>o.status===s).length}</span>}
          </button>
        ))}
        <input
          placeholder="🔍 Cari nama pelanggan..."
          value={searchOrder}
          onChange={e => setSearchOrder(e.target.value)}
          style={{ marginLeft: 'auto', padding: '7px 14px', borderRadius: 20, border: '1.5px solid #EEE4E1', fontSize: 12, fontFamily: 'Poppins,sans-serif', outline: 'none', minWidth: 200, background: '#fff' }}
        />
      </div>

      {/* Order Cards */}
      {filteredOrders.length === 0 ? (
        <div style={{ ...card, padding: '80px 20px', textAlign: 'center' }}>
          <div style={{ fontSize: 48, marginBottom: 12 }}>📭</div>
          <div style={{ fontSize: 15, fontWeight: 700, color: '#888' }}>Tidak ada pesanan</div>
          <div style={{ fontSize: 12, color: '#bbb', marginTop: 6 }}>Coba ubah filter atau tunggu pesanan baru</div>
        </div>
      ) : filteredOrders.map(order => {
        const sc  = STATUS[order.status] || STATUS.Menunggu
        const isUp = updatingId === order.id
        return (
          <div key={order.id} style={{ ...card, marginBottom: 14 }}>
            {/* Top Row */}
            <div style={{ padding: '18px 20px', display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start', gap: 12 }}>
              <div style={{ flex: 1, minWidth: 0 }}>
                <div style={{ display: 'flex', alignItems: 'center', gap: 8, marginBottom: 3 }}>
                  <div style={{ fontSize: 15, fontWeight: 700, color: '#1A0A0A' }}>{order.nama_pembeli || 'Pelanggan'}</div>
                  <span style={{ display: 'inline-flex', alignItems: 'center', gap: 4, background: sc.bg, color: sc.text, fontSize: 10, fontWeight: 700, padding: '3px 10px', borderRadius: 20 }}>
                    <span style={{ width: 5, height: 5, borderRadius: '50%', background: sc.dot, display: 'inline-block' }} />{order.status}
                  </span>
                </div>
                <div style={{ fontSize: 10, color: '#bbb', fontFamily: 'monospace', marginBottom: 6 }}>#{order.id?.slice(0, 8).toUpperCase()} · {fmtDate(order.created_at)}</div>
                <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>
                  {order.alamat && <span style={{ fontSize: 11, color: '#888' }}>📍 {order.alamat}</span>}
                  <span style={{ fontSize: 11, color: '#888' }}>💳 {order.metode_bayar === 'qris' ? 'QRIS/Transfer' : 'Bayar Langsung'}</span>
                </div>
                {order.metode_bayar === 'qris' && order.bukti_transfer && (
                  <a href={order.bukti_transfer} target="_blank" rel="noreferrer"
                    style={{ display: 'inline-block', marginTop: 6, fontSize: 11, color: M, fontWeight: 700, textDecoration: 'none', background: '#FFF4F4', padding: '3px 10px', borderRadius: 8 }}>
                    📎 Lihat Bukti Transfer ↗
                  </a>
                )}
              </div>
              <div style={{ textAlign: 'right', flexShrink: 0 }}>
                <div style={{ fontSize: 20, fontWeight: 900, color: M, letterSpacing: '-0.5px' }}>{fmtRp(order.total_harga)}</div>
              </div>
            </div>

            {/* Items */}
            {order.items?.length > 0 && (
              <div style={{ padding: '0 20px 14px', display: 'flex', flexWrap: 'wrap', gap: 6 }}>
                {order.items.map((item, i) => (
                  <span key={i} style={{ background: '#F5EEEE', color: M, fontSize: 11, fontWeight: 600, padding: '4px 10px', borderRadius: 20 }}>
                    {item.name} ×{item.quantity}
                  </span>
                ))}
              </div>
            )}

            {/* Status Buttons */}
            <div style={{ borderTop: '1px solid #F5EEEE', padding: '12px 16px', display: 'flex', gap: 8 }}>
              <span style={{ fontSize: 11, color: '#aaa', fontWeight: 600, alignSelf: 'center', marginRight: 4 }}>Update:</span>
              {FLOW.map(s => {
                const active = order.status === s
                const c = STATUS[s]
                return (
                  <button key={s} onClick={() => updateStatus(order.id, s)} disabled={isUp}
                    style={{ padding: '7px 14px', borderRadius: 20, fontSize: 11, fontWeight: 700, fontFamily: 'Poppins,sans-serif', cursor: isUp ? 'not-allowed' : 'pointer', border: active ? `2px solid ${c.dot}` : '2px solid transparent', background: active ? c.bg : '#F5F5F5', color: active ? c.text : '#aaa', opacity: isUp ? 0.5 : 1, transition: 'all 0.2s' }}>
                    {c.icon} {s}
                  </button>
                )
              })}
            </div>
          </div>
        )
      })}
    </div>
  )

  // MENU VIEW
  const MenuView = () => (
    <div>

]

  const router = useRouter()

  const [type, setType] = useState('delivery')
  const [nama, setNama] = useState('')
  const [alamat, setAlamat] = useState('')
  const [payment, setPayment] = useState('cash') // 'cash' atau 'qris'
  const [file, setFile] = useState(null)
  const [loading, setLoading] = useState(false)
  const [voucher, setVoucher] = useState(VOUCHERS[0])
  const [showVoucher, setShowVoucher] = useState(false)
  const [delModal, setDelModal] = useState(null)

  useEffect(() => {
    const c = JSON.parse(localStorage.getItem('cart') || '[]')
    setTimeout(() => setCart(c), 0)
  }, [])

  const save = c => { setCart(c); localStorage.setItem('cart', JSON.stringify(c)) }
  const inc = id => save(cart.map(i => i.id === id ? { ...i, quantity: i.quantity + 1 } : i))
  const dec = id => {
    const item = cart.find(i => i.id === id)
    if (!item) return
    if (item.quantity === 1) setDelModal(item)
    else save(cart.map(i => i.id === id ? { ...i, quantity: i.quantity - 1 } : i))
  }

  const total = sub - disc

    if (!nama.trim()) return alert('Masukkan nama kamu!')

      const { data, error } = await supabase.storage.from('bukti-transfer').upload(fname, file)
      
      if (!error) {
        buktiUrl = supabase.storage.from('bukti-transfer').getPublicUrl(fname).data.publicUrl
      }
    }

      (buktiUrl ? `📎 *Bukti Transfer:* ${buktiUrl}\n` : '') +
      `Mohon konfirmasi pesanan saya 🙏`

      // 1. Ambil ID Pesanan yang ada di HP user ini
      const myOrders = JSON.parse(localStorage.getItem('my_orders') || '[]')
      
      // Kalau nggak pernah pesen, kosongin aja nggak usah nanya ke database
      if (myOrders.length === 0) {
        setOrders([])
        setLoading(false)
        return
      }

      const { data } = await supabase
        .from('orders')
        .select('*')
        .in('id', myOrders)

      setOrders(data || [])
      setLoading(false)
    }

  }, [])

import Link from 'next/link'
import { usePathname } from 'next/navigation'

const NAV = [
  { href: '/', label: 'Home', icon: '/icons/home.svg', iconSel: '/icons/homeSelect.svg' },
  { href: '/menu', label: 'Menu', icon: '/icons/menu.svg', iconSel: '/icons/menuSelect.svg' },
  { href: '/history', label: 'Pesanan', icon: '/icons/order.svg', iconSel: '/icons/orderSelect.svg' },
]

export default function NavBar() {
  const pathname = usePathname()
  const hide = ['/cart', '/admin', '/login']
  if (hide.some(r => pathname.startsWith(r)) || /^\/menu\/.+/.test(pathname)) return null

  return (
    <div style={{ position: 'fixed', bottom: '16px', left: '50%', transform: 'translateX(-50%)', zIndex: 1000 }}>
      <nav style={{
        display: 'flex', alignItems: 'center',
        background: '#8B1A1A',
        borderRadius: '50px',
        padding: '8px 16px',
        gap: '4px',
        boxShadow: '0 8px 32px rgba(139,26,26,0.35), 0 2px 8px rgba(0,0,0,0.1)',
      }}>
        {NAV.map(({ href, label, icon, iconSel }) => {
          const active = pathname === href || (href === '/menu' && pathname === '/menu')
          if (active) return (
            <Link key={href} href={href} style={{
              display: 'flex', alignItems: 'center', justifyContent: 'center',
              background: '#fff',
              borderRadius: '50%',
              width: '52px', height: '52px', minWidth: '52px',

            }}>
              <img src={iconSel} alt={label} width={24} height={24} style={{ objectFit: 'contain' }} />
